# 数组的定义和访问
```cpp
//数据类型 数组名[数组长度];
//变量名必须以下划线或者字母开头
void test01(){
	int a[5];//用来储存int型变量的数组，长度为5
	//访问数组中的元素 数组名[下标];    数组的下标以0开始
	a[0] = 1;
	a[1] = 2;
	a[4] = 5;
	for(int i = 0;i < 5;i++){
		cout<<a[i]<<endl;
	}
}
//数据类型 数组名[数组长度] = {...};
//数据类型 数组名[] = {...};
void test02(){
	int a[] = {1,2,3,4,5};
	for(int i = 0;i < 5;i++){
		cout<<a[i]<<endl;
	}
	int b[5] = {1,2,3};//后面的元素自动补0
	for(int i = 0;i < 5;i++){
		cout<<b[i]<<endl;
	}
}
//数据类型 数组名[长度] = {number};
void test03(){
	int a[5] = {0};
	for(int i = 0;i < 5;i++){
		cout<<a[i]<<endl;
	}
}
```

//char str[] = "hello world"; //已经自动补了'\0'

}
