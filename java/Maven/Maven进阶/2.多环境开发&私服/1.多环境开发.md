## 多环境开发
- maven提供了多种环境的设定，帮助开发者使用过程中快速切换环境
`多环境的配置`
```xml
<profiles>
	<!--定义开发环境-->
	<profile>
		<id>env_dep</id>
		<properties>
			<jdbc.url>jdbc:mysql://127.0.0.1:3306/db1</jdbc.url>
		</properties>
		<!--设定是否为默认启动的环境-->
		<activation>
			<activeByDefault>true</activeByDefault>
		</activation>
	</profile>
		<!--定义测试环境-->
	<profile>
		<id>env_test</id>
		<properties>
			<jdbc.url>jdbc:mysql://127.2.0.1:3306/db1</jdbc.url>
		</properties>
	</profile>
		<!--定义生产环境-->
	<profile>
		<id>env_pro</id>
		<properties>
			<jdbc.url>jdbc:mysql://127.3.0.1:3306/db1</jdbc.url>
		</properties>
	</profile>
</profiles>
```
也可以使用指令指定使用哪一个环境：
```cmd
mvn install -P env_dep
```
